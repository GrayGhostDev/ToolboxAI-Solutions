    üöÄ Development Setup¶

    Prerequisites Check¶

    # Verify all required tools
docker --version          # Should be 25.x+
docker compose version    # Should be v2.x+
python3 --version        # Should be 3.12+
node --version           # Should be v22+
npm --version            # Should be 10+
git --version            # Should be 2.40+

    Quick Start (Docker - Recommended)¶

    # Clone and navigate
git clone https://github.com/ToolBoxAI-Solutions/toolboxai.git
cd toolboxai

# Start all services
docker compose -f infrastructure/docker/compose/docker-compose.yml \
               -f infrastructure/docker/compose/docker-compose.dev.yml up -d

# Check status
docker compose ps

# View logs
docker compose logs -f

    Quick Start (Native - Internal Drives Only)¶

    # Backend
cd apps/backend
python3 -m venv venv
source venv/bin/activate
pip install -r requirements.txt
uvicorn main:app --host 127.0.0.1 --port 8009 --reload

# Frontend (new terminal)
cd apps/dashboard
npm install
npm run dev  # Starts on port 5179

    üê≥ Docker Commands¶

    Service Management¶

    # Start all services
docker compose up -d

# Stop all services
docker compose down

# Restart specific service
docker compose restart fastapi-backend

# View service status
docker compose ps

# View logs (all services)
docker compose logs -f

# View logs (specific service)
docker compose logs -f dashboard-frontend

# Shell into container
docker compose exec fastapi-backend /bin/bash
docker compose exec dashboard-frontend /bin/sh

    Development Workflow¶

    # Rebuild after code changes
docker compose up -d --build

# Rebuild specific service
docker compose up -d --build fastapi-backend

# Clean rebuild (removes volumes)
docker compose down -v
docker compose up -d --build

# View resource usage
docker stats

# Clean up unused resources
docker system prune -a --volumes

    Troubleshooting¶

    # Check container health
docker compose ps --format "table {{.Name}}\t{{.Status}}\t{{.Ports}}"

# Inspect container
docker inspect toolboxai-fastapi-backend

# Check networks
docker network ls
docker network inspect toolboxai-network

# View container logs (last 100 lines)
docker compose logs --tail=100 fastapi-backend

    üêç Python/Backend¶

    Virtual Environment¶

    # Create venv
python3 -m venv venv

# Activate (macOS/Linux)
source venv/bin/activate

# Activate (Windows)
venv\Scripts\activate

# Deactivate
deactivate

# Install dependencies
pip install -r requirements.txt

# Update dependencies
pip install --upgrade -r requirements.txt

    FastAPI Development¶

    # Start development server
cd apps/backend
uvicorn main:app --host 127.0.0.1 --port 8009 --reload

# With auto-reload on file changes
uvicorn main:app --reload --reload-dir . --reload-exclude "*.pyc"

# Check OpenAPI docs
open http://127.0.0.1:8009/docs          # Swagger UI
open http://127.0.0.1:8009/redoc         # ReDoc

# Interactive shell
python -i -c "from main import app; from toolboxai_settings import settings"

    Code Quality¶

    # Type checking (BasedPyright)
basedpyright .

# Format code (Black)
black apps/backend

# Lint code (Ruff)
ruff check apps/backend

# Fix linting issues
ruff check --fix apps/backend

# Check imports (isort)
isort --check-only apps/backend
isort apps/backend  # Fix

    Database Migrations¶

    # Create migration
cd apps/backend
alembic revision --autogenerate -m "Description of changes"

# Apply migration
alembic upgrade head

# Rollback one migration
alembic downgrade -1

# View migration history
alembic history

# Current database version
alembic current

    üì¶ TypeScript/Frontend¶

    Package Management¶

    cd apps/dashboard

# Install dependencies
npm install

# External drive workaround (if needed)
npm install --no-bin-links --legacy-peer-deps

# Add new dependency
npm install <package>

# Add dev dependency
npm install -D <package>

# Update dependencies
npm update

# Audit dependencies
npm audit
npm audit fix

    Development Server¶

    # Start Vite dev server
npm run dev              # Default port 5179

# Custom port
PORT=5180 npm run dev

# Network accessible
npm run dev -- --host 0.0.0.0

# Open in browser
npm run dev -- --open

    Build & Deploy¶

    # Build for production
npm run build

# Preview production build
npm run preview

# Build with type checking
npm run typecheck && npm run build

# Build with linting
npm run lint && npm run build

    Code Quality¶

    # ESLint 9 (Flat Config)
npm run lint
npm run lint:fix

# TypeScript type checking
npm run typecheck

# Format code (Prettier, if configured)
npm run format
npm run format:check

    üß™ Testing¶

    Python Tests¶

    # Run all tests
pytest

# Run with verbose output
pytest -v

# Run specific test file
pytest tests/unit/test_api.py

# Run specific test
pytest tests/unit/test_api.py::test_login

# Run with coverage
pytest --cov=apps/backend --cov-report=html

# Run tests matching pattern
pytest -k "auth"

# Run unit tests only
pytest -m unit

# Run integration tests
pytest -m integration

# Environment flags for gated tests
RUN_ENDPOINT_TESTS=1 pytest
RUN_ROJO_TESTS=1 pytest

    Frontend Tests¶

    cd apps/dashboard

# Run all tests (Vitest)
npm test

# Run in watch mode
npm test -- --watch

# Run with coverage
npm run test:coverage

# Run specific test file
npm test -- src/__tests__/components/Header.test.tsx

# Run tests matching pattern
npm test -- --grep "authentication"

# UI mode (interactive)
npm test -- --ui

    üóÑÔ∏è Database¶

    PostgreSQL¶

    # Connect to database (Docker)
docker compose exec postgres psql -U eduplatform -d educational_platform_dev

# Connect to database (native)
psql -h localhost -p 5434 -U eduplatform -d educational_platform_dev

# Common SQL commands
\l                 # List databases
\c database_name   # Connect to database
\dt                # List tables
\d table_name      # Describe table
\q                 # Quit

# Backup database
pg_dump -h localhost -p 5434 -U eduplatform educational_platform_dev > backup.sql

# Restore database
psql -h localhost -p 5434 -U eduplatform educational_platform_dev < backup.sql

    Redis¶

    # Connect to Redis (Docker)
docker compose exec redis redis-cli

# Connect to Redis (native)
redis-cli -p 6381

# Common Redis commands
KEYS *             # List all keys
GET key_name       # Get value
SET key value      # Set value
DEL key_name       # Delete key
FLUSHDB            # Clear database
FLUSHALL           # Clear all databases
INFO               # Server info

    üîÄ Git Workflow¶

    Daily Development¶

    # Update main branch
git checkout main
git pull origin main

# Create feature branch
git checkout -b feature/my-feature

# Stage changes
git add .
git status

# Commit with message
git commit -m "feat: add new feature"

# Push to remote
git push origin feature/my-feature

# Create pull request (via GitHub CLI)
gh pr create --title "Add new feature" --body "Description"

    Conventional Commits¶

    # Feature
git commit -m "feat: add user authentication"

# Bug fix
git commit -m "fix: resolve login issue"

# Documentation
git commit -m "docs: update API documentation"

# Refactoring
git commit -m "refactor: improve code structure"

# Tests
git commit -m "test: add unit tests for auth"

# Chore
git commit -m "chore: update dependencies"

# Breaking change
git commit -m "feat!: change API endpoint structure

BREAKING CHANGE: /api/v1/users renamed to /api/v2/users"

    Useful Git Commands¶

    # View recent commits
git log --oneline -10

# View file history
git log --follow -- path/to/file

# Undo last commit (keep changes)
git reset --soft HEAD~1

# Stash changes
git stash
git stash list
git stash apply

# Cherry-pick commit
git cherry-pick <commit-hash>

# Clean untracked files
git clean -fd

    üîß Troubleshooting¶

    Port Conflicts¶

    # Find process using port
lsof -i :8009  # Backend
lsof -i :5179  # Frontend

# Kill process
kill -9 <PID>

# Use alternative ports
PORT=8010 uvicorn main:app  # Backend
PORT=5180 npm run dev       # Frontend

    External Drive Issues (macOS)¶

    # Issue: Native binaries fail with error -88
# Solution 1: Use Docker (recommended)
docker compose up -d

# Solution 2: NPM workarounds
cd apps/dashboard
npm install --no-bin-links --legacy-peer-deps

# Solution 3: Use JavaScript config
# Rename vite.config.ts to vite.config.js
mv vite.config.ts vite.config.js

    Docker Issues¶

    # Container won't start
docker compose logs <service-name>
docker compose ps

# Permission issues (non-root users)
sudo chown -R 1001:1001 ./data/postgres/
sudo chown -R 1002:1002 ./data/redis/

# Network issues
docker network ls
docker network prune

# Rebuild from scratch
docker compose down -v
docker compose build --no-cache
docker compose up -d

    Python Issues¶

    # Module not found
pip install -r requirements.txt
pip list  # Verify installation

# Type checking errors
basedpyright .

# Import errors
python -c "import module_name"  # Test import

# Virtual environment not activated
source venv/bin/activate  # Activate first

    Frontend Issues¶

    # Dependencies not installing
rm -rf node_modules package-lock.json
npm install --legacy-peer-deps

# Build failures
npm run typecheck  # Check types first
npm run lint       # Check linting

# Hot reload not working
# Check vite.config.js server.watch settings

# CORS errors
# Check backend CORS configuration in main.py
