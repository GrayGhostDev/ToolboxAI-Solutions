    Obtain JWT Token¶

    import httpx
from pydantic import BaseModel

class LoginRequest(BaseModel):
    email: str
    password: str

class LoginResponse(BaseModel):
    access_token: str
    token_type: str
    expires_in: int

async def login(email: str, password: str) -> str:
    """Login and return JWT token."""
    async with httpx.AsyncClient() as client:
        response = await client.post(
            "http://localhost:8009/api/v1/auth/login",
            json={"email": email, "password": password}
        )
        response.raise_for_status()
        data = LoginResponse.model_validate(response.json())
        return data.access_token

# Usage
token = await login("user@example.com", "password123")

    interface LoginRequest {
  email: string;
  password: string;
}

interface LoginResponse {
  access_token: string;
  token_type: string;
  expires_in: number;
}

async function login(email: string, password: string): Promise<string> {
  const response = await fetch('http://localhost:8009/api/v1/auth/login', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ email, password })
  });

  if (!response.ok) {
    throw new Error(`Login failed: ${response.statusText}`);
  }

  const data: LoginResponse = await response.json();
  return data.access_token;
}

// Usage
const token = await login('user@example.com', 'password123');

    # Login to get JWT token
curl -X POST http://localhost:8009/api/v1/auth/login \
  -H "Content-Type: application/json" \
  -d '{
    "email": "user@example.com",
    "password": "password123"
  }'

# Response:
# {
#   "access_token": "eyJhbGciOiJIUzI1NiIs...",
#   "token_type": "bearer",
#   "expires_in": 3600
# }

    Get Current User Profile¶

    from typing import Annotated
from pydantic import BaseModel, EmailStr

class UserProfile(BaseModel):
    id: int
    email: EmailStr
    full_name: str | None = None
    role: str
    created_at: str

async def get_current_user(token: str) -> UserProfile:
    """Get current user profile."""
    async with httpx.AsyncClient() as client:
        response = await client.get(
            "http://localhost:8009/api/v1/users/me",
            headers={"Authorization": f"Bearer {token}"}
        )
        response.raise_for_status()
        return UserProfile.model_validate(response.json())

# Usage
user = await get_current_user(token)
print(f"Logged in as: {user.email} (Role: {user.role})")

    interface UserProfile {
  id: number;
  email: string;
  full_name: string | null;
  role: string;
  created_at: string;
}

async function getCurrentUser(token: string): Promise<UserProfile> {
  const response = await fetch('http://localhost:8009/api/v1/users/me', {
    headers: { 'Authorization': `Bearer ${token}` }
  });

  if (!response.ok) {
    throw new Error(`Failed to get user: ${response.statusText}`);
  }

  return await response.json();
}

// Usage
const user = await getCurrentUser(token);
console.log(`Logged in as: ${user.email} (Role: ${user.role})`);

    Update User Profile¶

    from pydantic import BaseModel

class UserUpdate(BaseModel):
    full_name: str | None = None
    bio: str | None = None

async def update_profile(
    token: str,
    updates: UserUpdate
) -> UserProfile:
    """Update user profile."""
    async with httpx.AsyncClient() as client:
        response = await client.patch(
            "http://localhost:8009/api/v1/users/me/profile",
            headers={"Authorization": f"Bearer {token}"},
            json=updates.model_dump(exclude_none=True)
        )
        response.raise_for_status()
        return UserProfile.model_validate(response.json())

# Usage
updated = await update_profile(
    token,
    UserUpdate(full_name="Jane Smith", bio="Teacher at XYZ School")
)

    interface UserUpdate {
  full_name?: string;
  bio?: string;
}

async function updateProfile(
  token: string,
  updates: UserUpdate
): Promise<UserProfile> {
  const response = await fetch(
    'http://localhost:8009/api/v1/users/me/profile',
    {
      method: 'PATCH',
      headers: {
        'Authorization': `Bearer ${token}`,
        'Content-Type': 'application/json'
      },
      body: JSON.stringify(updates)
    }
  );

  if (!response.ok) {
    throw new Error(`Update failed: ${response.statusText}`);
  }

  return await response.json();
}

// Usage
const updated = await updateProfile(token, {
  full_name: 'Jane Smith',
  bio: 'Teacher at XYZ School'
});

    # Update user profile
curl -X PATCH http://localhost:8009/api/v1/users/me/profile \
  -H "Authorization: Bearer $TOKEN" \
  -H "Content-Type: application/json" \
  -d '{
    "full_name": "Jane Smith",
    "bio": "Teacher at XYZ School"
  }'

    List Educational Content¶

    from pydantic import BaseModel

class ContentItem(BaseModel):
    id: int
    title: str
    content_type: str
    difficulty: str
    created_at: str

class ContentListResponse(BaseModel):
    items: list[ContentItem]
    total: int
    page: int
    page_size: int

async def list_content(
    token: str,
    page: int = 1,
    page_size: int = 20,
    difficulty: str | None = None
) -> ContentListResponse:
    """List educational content with pagination."""
    params = {"page": page, "page_size": page_size}
    if difficulty:
        params["difficulty"] = difficulty

    async with httpx.AsyncClient() as client:
        response = await client.get(
            "http://localhost:8009/api/v1/content",
            headers={"Authorization": f"Bearer {token}"},
            params=params
        )
        response.raise_for_status()
        return ContentListResponse.model_validate(response.json())

# Usage
content = await list_content(token, page=1, difficulty="beginner")
print(f"Found {content.total} items (showing {len(content.items)})")

    interface ContentItem {
  id: number;
  title: string;
  content_type: string;
  difficulty: string;
  created_at: string;
}

interface ContentListResponse {
  items: ContentItem[];
  total: number;
  page: number;
  page_size: number;
}

async function listContent(
  token: string,
  options: {
    page?: number;
    page_size?: number;
    difficulty?: string;
  } = {}
): Promise<ContentListResponse> {
  const params = new URLSearchParams({
    page: String(options.page ?? 1),
    page_size: String(options.page_size ?? 20),
    ...(options.difficulty && { difficulty: options.difficulty })
  });

  const response = await fetch(
    `http://localhost:8009/api/v1/content?${params}`,
    {
      headers: { 'Authorization': `Bearer ${token}` }
    }
  );

  if (!response.ok) {
    throw new Error(`Failed to list content: ${response.statusText}`);
  }

  return await response.json();
}

// Usage
const content = await listContent(token, { page: 1, difficulty: 'beginner' });
console.log(`Found ${content.total} items (showing ${content.items.length})`);

    # List content with pagination and filtering
curl -X GET "http://localhost:8009/api/v1/content?page=1&page_size=20&difficulty=beginner" \
  -H "Authorization: Bearer $TOKEN"

#     }
#   ],

    Subscribe to Content Generation Events¶

    import pusher

# Initialize Pusher client
pusher_client = pusher.Client(
    app_id='your-app-id',
    key='your-key',
    secret='your-secret',
    cluster='your-cluster',
    ssl=True
)

# Trigger event
pusher_client.trigger(
    'content-generation',
    'generation-progress',
    {
        'content_id': 123,
        'progress': 75,
        'message': 'Generating quiz questions...'
    }
)

    import Pusher from 'pusher-js';

// Initialize Pusher client
const pusher = new Pusher('your-key', {
  cluster: 'your-cluster',
  authEndpoint: 'http://localhost:8009/pusher/auth',
  auth: {
    headers: {
      'Authorization': `Bearer ${token}`
    }
  }
});

// Subscribe to channel
const channel = pusher.subscribe('content-generation');

// Listen for events
channel.bind('generation-progress', (data: {
  content_id: number;
  progress: number;
  message: string;
}) => {
  console.log(`Progress: ${data.progress}% - ${data.message}`);
});

// Unsubscribe when done
// channel.unbind_all();
// pusher.unsubscribe('content-generation');

    import { usePusherEvent } from '@/hooks/pusher';

function ContentGenerator() {
  const [progress, setProgress] = useState(0);
  const [message, setMessage] = useState('');

  usePusherEvent('content-generation', 'generation-progress', (data) => {
    setProgress(data.progress);
    setMessage(data.message);
  });

  return (
    <div>
      <progress value={progress} max={100} />
      <p>{message}</p>
    </div>
  );
}

    Upload Educational Material¶

    from pathlib import Path

async def upload_file(
    token: str,
    file_path: Path,
    content_type: str = "lesson"
) -> dict[str, str]:
    """Upload educational material file."""
    async with httpx.AsyncClient() as client:
        with open(file_path, 'rb') as f:
            files = {'file': (file_path.name, f, 'application/pdf')}
            data = {'content_type': content_type}

            response = await client.post(
                "http://localhost:8009/api/v1/content/upload",
                headers={"Authorization": f"Bearer {token}"},
                files=files,
                data=data
            )
            response.raise_for_status()
            return response.json()

# Usage
result = await upload_file(
    token,
    Path("lesson_materials.pdf"),
    content_type="lesson"
)
print(f"Uploaded: {result['file_url']}")

    async function uploadFile(
  token: string,
  file: File,
  contentType: string = 'lesson'
): Promise<{ file_url: string; file_id: string }> {
  const formData = new FormData();
  formData.append('file', file);
  formData.append('content_type', contentType);

  const response = await fetch(
    'http://localhost:8009/api/v1/content/upload',
    {
      method: 'POST',
      headers: { 'Authorization': `Bearer ${token}` },
      body: formData
    }
  );

  if (!response.ok) {
    throw new Error(`Upload failed: ${response.statusText}`);
  }

  return await response.json();
}

// Usage with file input
const input = document.querySelector<HTMLInputElement>('#file-input')!;
const file = input.files?.[0];
if (file) {
  const result = await uploadFile(token, file, 'lesson');
  console.log(`Uploaded: ${result.file_url}`);
}

    # Upload file
curl -X POST http://localhost:8009/api/v1/content/upload \
  -H "Authorization: Bearer $TOKEN" \
  -F "file=@lesson_materials.pdf" \
  -F "content_type=lesson"

# Response:
# {
#   "file_url": "https://storage.example.com/files/lesson_materials.pdf",
#   "file_id": "abc123"

    Proper Error Handling (2025 Patterns)¶

    from httpx import HTTPError, HTTPStatusError

async def safe_api_call(token: str) -> UserProfile | None:
    """API call with proper error handling."""
    try:
        async with httpx.AsyncClient() as client:
            response = await client.get(
                "http://localhost:8009/api/v1/users/me",
                headers={"Authorization": f"Bearer {token}"},
                timeout=10.0
            )
            response.raise_for_status()
            return UserProfile.model_validate(response.json())

    except HTTPStatusError as e:
        if e.response.status_code == 401:
            print("Authentication failed - token expired")
        elif e.response.status_code == 404:
            print("User not found")
        else:
            print(f"HTTP error: {e.response.status_code}")
        return None

    except HTTPError as e:
        print(f"Request failed: {e}")
        return None

    except Exception as e:
        print(f"Unexpected error: {e}")
        return None

# Usage
user = await safe_api_call(token)
if user:
    print(f"Success: {user.email}")
else:
    print("Failed to get user")

    async function safeApiCall(token: string): Promise<UserProfile | null> {
  try {
    const response = await fetch(
      'http://localhost:8009/api/v1/users/me',
      {
        headers: { 'Authorization': `Bearer ${token}` },
        signal: AbortSignal.timeout(10000) // 10s timeout
      }
    );

    if (!response.ok) {
      if (response.status === 401) {
        console.error('Authentication failed - token expired');
      } else if (response.status === 404) {
        console.error('User not found');
      } else {
        console.error(`HTTP error: ${response.status}`);
      }
      return null;
    }

    return await response.json();

  } catch (error) {
    if (error instanceof TypeError) {
      console.error('Network error:', error.message);
    } else {
      console.error('Unexpected error:', error);
    }
    return null;
  }
}

// Usage
const user = await safeApiCall(token);
if (user) {
  console.log(`Success: ${user.email}`);
} else {
  console.log('Failed to get user');
}
