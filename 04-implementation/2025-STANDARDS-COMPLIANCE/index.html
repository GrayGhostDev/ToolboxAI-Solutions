    ‚öõÔ∏è React 19.1.0 Standards¶

    ‚úÖ Correct Patterns (2025)¶

    Functional Components¶

    // ‚úÖ 2025: Function component with hooks
import { useState, useEffect } from 'react';

interface UserProps {
  userId: string;
}

export function UserProfile({ userId }: UserProps) {
  const [user, setUser] = useState<User | null>(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    fetchUser(userId).then(setUser).finally(() => setLoading(false));
  }, [userId]);

  if (loading) return <div>Loading...</div>;
  if (!user) return <div>User not found</div>;

  return (
    <div>
      <h1>{user.name}</h1>
      <p>{user.email}</p>
    </div>
  );
}

    ‚ùå Deprecated Patterns¶

    // ‚ùå WRONG: any type

  return data;
}

// ‚úÖ CORRECT: unknown with type guards
function processData(data: unknown) {
  if (typeof data === 'object' && data !== null) {
    return data;
  }
  throw new Error('Invalid data');
}

// ‚ùå WRONG: Non-null assertion overuse

// ‚úÖ CORRECT: Proper null checking
const user = users.find(u => u.id === id);

    Type Hints (PEP 484, 585, 604)¶

    # ‚úÖ 2025: Modern type hints (Python 3.12)
from typing import override

def process_users(users: list[dict[str, str | int]]) -> list[str]:
    """Process users with modern type hints."""
    return [user['name'] for user in users if isinstance(user['name'], str)]

# ‚úÖ 2025: Generic type alias (PEP 695)
type Vector[T] = list[T]
type Matrix[T] = list[Vector[T]]

# ‚úÖ 2025: Override decorator (PEP 698)
class UserService(BaseService):

from typing import AsyncIterator

async def stream_logs() -> AsyncIterator[str]:

    # ‚ùå WRONG: Old-style type hints

    üöÄ FastAPI Standards¶

    ‚úÖ Correct Patterns (2025)¶

    Route Definition¶

    # ‚úÖ 2025: Async endpoints with dependency injection
from fastapi import FastAPI, Depends, HTTPException, status
from pydantic import BaseModel

app = FastAPI()

class UserCreate(BaseModel):
    email: str
    password: str

class UserResponse(BaseModel):
    id: int
    email: str
    created_at: datetime

@app.post(
    "/users",
    response_model=UserResponse,
    status_code=status.HTTP_201_CREATED
)
async def create_user(
    user: UserCreate,
    db: AsyncSession = Depends(get_db)
) -> UserResponse:
    """Create a new user."""
    db_user = await UserService(db).create_user(user)
    return UserResponse.model_validate(db_user)

    Dependency Injection¶

    # ‚úÖ 2025: Proper dependency management
from typing import Annotated

async def get_current_user(
    token: Annotated[str, Depends(oauth2_scheme)],
    db: Annotated[AsyncSession, Depends(get_db)]
) -> User:
    """Get current authenticated user."""
    credentials_exception = HTTPException(
        status_code=status.HTTP_401_UNAUTHORIZED,
        detail="Could not validate credentials"
    )
    try:
        payload = jwt.decode(token, SECRET_KEY, algorithms=[ALGORITHM])
        user_id: str = payload.get("sub")
        if user_id is None:
            raise credentials_exception
    except JWTError:
        raise credentials_exception

    user = await db.get(User, user_id)
    if user is None:
        raise credentials_exception
    return user

    üì¶ Pydantic v2 Standards¶

    ‚úÖ Correct Patterns (2025)¶

    # ‚úÖ 2025: Pydantic v2 with ConfigDict
from pydantic import BaseModel, Field, field_validator, ConfigDict

class User(BaseModel):
    model_config = ConfigDict(
        str_strip_whitespace=True,
        validate_assignment=True,
        arbitrary_types_allowed=True
    )

    id: int
    email: str = Field(..., description="User email address")
    age: int = Field(ge=0, le=120)

    @field_validator('email')
    @classmethod
    def validate_email(cls, v: str) -> str:
        """Validate email format."""
        if '@' not in v:
            raise ValueError('Invalid email address')
        return v.lower()

    ‚ùå Deprecated Patterns¶

    # ‚ùå WRONG: Pydantic v1 patterns
class User(BaseModel):
    email: str

    class Config:  # ‚ùå v1 pattern

    üîç Code Review Checklist¶

    Before Committing Code¶

    • All type hints use modern syntax (Python 3.12, TS 5.9)
    • No any types in TypeScript
    • No bare except clauses in Python
    • React components are functional (no classes)
    • Pydantic models use v2 patterns
    • FastAPI routes use async/await
    • Vite config uses modern plugins
    • All imports reference official 2025 docs
    • No deprecated API usage
    • Code passes type checker (basedpyright, tsc)
